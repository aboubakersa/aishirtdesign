        // --- CHART RENDERING ---
        function renderRevenueChart(dateMap) {
            const ctx = document.getElementById('revenueChart').getContext('2d');
            const sortedDates = Object.keys(dateMap).sort((a,b) => new Date(a) - new Date(b));
            if(sortedDates.length === 0) sortedDates.push(new Date().toLocaleDateString());
            const dVals = sortedDates.map(l => dateMap[l] || 0);

            let gradient = ctx.createLinearGradient(0, 0, 0, 400);
            gradient.addColorStop(0, 'rgba(0, 184, 230, 0.4)');
            gradient.addColorStop(1, 'rgba(0, 184, 230, 0.0)');

            if(revChartInstance) {
                revChartInstance.data.labels = sortedDates;
                revChartInstance.data.datasets[0].data = dVals;
                revChartInstance.update();
            } else {
                revChartInstance = new Chart(ctx, { 
                    type: 'line', 
                    data: { 
                        labels: sortedDates, 
                        datasets: [{ 
                            label: 'Revenue (DZD)', 
                            data: dVals, 
                            borderColor: '#00b8e6', 
                            backgroundColor: gradient, 
                            borderWidth: 3, 
                            fill: true, 
                            tension: 0.4,
                            pointBackgroundColor: '#ffffff',
                            pointBorderColor: '#00b8e6',
                            pointBorderWidth: 2,
                            pointRadius: 4
                        }] 
                    }, 
                    options: { 
                        responsive: true, 
                        maintainAspectRatio: false, 
                        plugins: { legend: { display: false } },
                        scales: { y: { beginAtZero: true, grid: { borderDash: [5, 5], color: '#e2e8f0' } }, x: { grid: { display: false } } }
                    } 
                });
            }
        }
