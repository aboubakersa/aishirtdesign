        function appendMessage(role, content, isHtml = false, customClass = '') {
            const row = document.createElement('div');
            row.className = `msg-row ${role}`;

            const avatar = document.createElement('div');
            avatar.className = 'msg-avatar';
            avatar.innerHTML = role === 'ai' ? 
                `<svg viewBox="0 0 24 24" fill="currentColor" width="22" height="22"><path d="M12,2A10,10 0 0,0 2,12A10,10 0 0,0 12,22A10,10 0 0,0 22,12A10,10 0 0,0 12,2M12,8.5C13.93,8.5 15.5,10.07 15.5,12C15.5,13.93 13.93,15.5 12,15.5C10.07,15.5 8.5,13.93 8.5,12C8.5,10.07 10.07,8.5 12,8.5M12,10.5C11.17,10.5 10.5,11.17 10.5,12C10.5,12.83 11.17,13.5 12,13.5C12.83,13.5 13.5,12.83 13.5,12C13.5,11.17 12.83,10.5 12,10.5Z"/></svg>` : 
                `<span>U</span>`;

            const contentWrapper = document.createElement('div');
            contentWrapper.className = 'msg-content-wrapper';
            
            const nameLabel = document.createElement('div');
            nameLabel.className = 'msg-name';
            nameLabel.innerText = role === 'ai' ? 'Tasmimak' : 'YOU';
            contentWrapper.appendChild(nameLabel);

            const bubble = document.createElement('div');
            bubble.className = `msg-bubble ${customClass}`;
            if (isHtml) bubble.innerHTML = content; else bubble.innerText = content;
            contentWrapper.appendChild(bubble);

            row.appendChild(avatar); 
            row.appendChild(contentWrapper); 

            chatWindow.appendChild(row);
            chatWindow.scrollTop = chatWindow.scrollHeight;
            return row;
        }

        let unsubHistory = null;
        function loadChatHistory() {
            if(!currentUser) return;
            if (unsubHistory) unsubHistory();
            
            unsubHistory = onSnapshot(collection(db, 'users', currentUser.uid, 'chats'), (snap) => {
                const chats = []; snap.forEach(doc => chats.push({id: doc.id, ...doc.data()}));
                chats.sort((a, b) => (b.timestamp?.toMillis() || Date.now()) - (a.timestamp?.toMillis() || Date.now()));

                chatHistoryList.innerHTML = chats.length ? '' : `<div style="text-align: center; color:var(--text-muted); font-size: 0.85rem; margin-top:20px; font-weight: 600;"><span class="hide-on-collapse">No recent concepts</span></div>`;
                
                chats.forEach(data => {
                    const item = document.createElement('div'); 
                    item.className = 'history-item';
                    item.innerHTML = `<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M21 15a2 2 0 0 1-2 2H7l-4 4V5a2 2 0 0 1 2-2h14a2 2 0 0 1 2 2z"></path></svg> <span class="hide-on-collapse">${data.prompt}</span>`;
                    
                    item.onclick = () => { 
                        chatWindow.innerHTML = '';

                                <img src="${data.imageUrl}">

        }
        
        // --- MY PRODUCTS FETCHING ---
        let unsubProducts = null;
        function loadUserProducts() {
            if(!currentUser) return;
            if (unsubProducts) unsubProducts();
            
            unsubProducts = onSnapshot(collection(db, 'users', currentUser.uid, 'orders'), (snap) => {
                const grid = document.getElementById('productsGridList'); 
                grid.innerHTML = '';
                let total = 0; let pending = 0;
                
                const orders = []; snap.forEach(doc => orders.push(doc.data()));
                orders.sort((a,b) => (b.timestamp?.toMillis() || 0) - (a.timestamp?.toMillis() || 0));

                orders.forEach(data => {
                    total++; if(data.status === 'Pending' || data.status === 'Processing') pending++;
                    const item = document.createElement('div'); 
                    item.className = 'product-item';
                    item.innerHTML = `
                        <img src="${data.imageUrl}" alt="Design">
                        <div class="product-meta"><span style="font-weight: 800; color: var(--text-dark);">${data.productType || 'T-Shirt'}</span></div>

                    `;
                    grid.appendChild(item);

        // --- AI GENERATION ENGINE (NODE.JS CALL) ---
        window.processUserRequest = async function() {
            if (isProcessing) return;
            if (localGenerationsLeft <= 0) return appendMessage('ai', '🛑 Limit Reached. Please upgrade your account in the Dashboard to continue generating.', true, 'msg-error');
            
            const prompt = promptInput.value.trim(); 
            if (!prompt) return;
            
            isProcessing = true; 
            promptInput.value = ''; 
            promptInput.disabled = true; 
            sendBtn.disabled = true;

            appendMessage('user', prompt, false);
            const loadingHtml = `
                <div class="status-text">Synthesizing design...</div>
                <div class="typing"><span></span><span></span><span></span></div>
            `;
            const loading = appendMessage('ai', loadingHtml, true);

            try {
                const idToken = await currentUser.getIdToken();

                const response = await fetch("http://localhost:3000/api/generate-design", {
                    method: "POST",
                    headers: { 
                        "Content-Type": "application/json",
                        "Authorization": `Bearer ${idToken}` 
                    },
                    body: JSON.stringify({ 
                        prompt: prompt,
                        product: currentDesignConfig.product,
                        quality: currentDesignConfig.quality,
                        material: currentDesignConfig.material
                    })
                });

                const data = await response.json();
                chatWindow.removeChild(loading);

                if (!response.ok) {
                    throw new Error(data.error || "Failed to generate from server.");
                }

                currentGeneratedImage = data.imageUrl;
                localGenerationsLeft = data.generationsLeft;
                document.getElementById('uiGenerationsCount').innerText = localGenerationsLeft;
                
                const finalPrice = currentDesignConfig.basePrice;

                appendMessage('ai', `
                    <div class="status-text">🎨 Rendering Complete</div>
                    <img src="${data.imageUrl}" alt="Generated Design" onload="document.getElementById('chatWindow').scrollTop = document.getElementById('chatWindow').scrollHeight">
                    
                    <div style="background: var(--bg-body); padding: 15px; border-radius: 14px; margin-top: 15px; display:flex; justify-content: space-between; align-items: center; border: 1px solid var(--border-light);">
                        <div style="display: flex; flex-direction: column;">
                            <span style="font-weight:800; color:var(--text-dark); font-size: 0.95rem;">${currentDesignConfig.product}</span>
                            <span style="font-weight:600; color:var(--text-muted); font-size: 0.8rem;">Includes AI Generation Fee (0.04)</span>
                        </div>
                        <span style="font-weight:800; font-size: 1.3rem; color: var(--brand-cyan-dark);">${finalPrice} DZD</span>
                    </div>

                    <div class="decision-actions">
                        <button class="btn-decision accept" onclick="window.prepareOrderModal('${data.imageUrl}')">
                            <svg viewBox="0 0 24 24" width="18" height="18" fill="none" stroke="currentColor" stroke-width="2"><path d="M22 11.08V12a10 10 0 1 1-5.93-9.14"></path><polyline points="22 4 12 14.01 9 11.01"></polyline></svg>
                            Order Design
                        </button>
                        <button class="btn-decision refuse" onclick="window.discardDesign()">
                            <svg viewBox="0 0 24 24" width="18" height="18" fill="none" stroke="currentColor" stroke-width="2"><path d="M3 6h18"></path><path d="M19 6v14a2 2 0 0 1-2 2H7a2 2 0 0 1-2-2V6m3 0V4a2 2 0 0 1 2-2h4a2 2 0 0 1 2 2v2"></path></svg>
                            Discard
                        </button>
                    </div>
                `, true);

            } catch (error) {
                console.error(error);
                if (chatWindow.contains(loading)) chatWindow.removeChild(loading);
                appendMessage('ai', `❌ Server Error: ${error.message}`, false, 'msg-error');
            }

            isProcessing = false; 
            promptInput.disabled = false; 
            sendBtn.disabled = false; 
            promptInput.focus();
        };

        window.discardDesign = function() {
            appendMessage('ai', 'Design discarded. Provide a new prompt to try again.', false, 'msg-system');
        };

        window.submitOrder = async function() {
            const size = document.getElementById('orderSize').value;
            const wilaya = document.getElementById('orderWilaya').value;
            const name = document.getElementById('orderName').value.trim();
            const phone = document.getElementById('orderPhone').value.trim();
            const address = document.getElementById('orderAddress').value.trim();
            
            if(!size || !wilaya || !name || !phone || !address) return showToast("Please fill all fields.", "error");
            
            const submitBtn = document.getElementById('confirmOrderBtn');
            const originalText = submitBtn.innerText;
            submitBtn.innerText = "Processing...";
            submitBtn.disabled = true;
            
            try {
                // Save Order into User's 'orders' subcollection
                await addDoc(collection(db, 'users', currentUser.uid, 'orders'), {
                    imageUrl: currentGeneratedImage, 
                    size: size, 
                    wilaya: wilaya, 
                    customerName: name, 
                    productType: currentDesignConfig.product, 
                    material: currentDesignConfig.material, 
                    price: currentDesignConfig.basePrice, 
                    phone: phone, 
                    address: address,
                    status: 'Pending', 
                    timestamp: serverTimestamp()
                });
                
                // Clear Form
                document.getElementById('orderSize').value = '';
                document.getElementById('orderWilaya').value = '';
                document.getElementById('orderName').value = '';
                document.getElementById('orderPhone').value = '';
                document.getElementById('orderAddress').value = '';

                closeOrderModal(); 
                document.getElementById('successModal').classList.add('active');
                
            } catch(e) { 
                showToast("Failed to submit order: " + e.message, "error"); 
            } finally {
                submitBtn.innerText = originalText;
                submitBtn.disabled = false;
            }
        };
